int reversed = 0;
int temp = x;
if (temp < 0) {
    temp = -temp;  
}
while (temp > 0) {
    reversed = reversed * 10 + temp % 10; 
    temp /= 10;
}

    reversed = -reversed;  
}
printf("反转结果：%d\n", reversed);
return 0;
