Leetcode-3最长不重复子串

# Leetcode-3最长不重复子串

#include <stdio.h>
int LongestUniqueSubstring(char *arr, int arrsize)
{
    if (arr == NULL || arrsize <= 0) {
        printf("输入字符串为空或长度为0\n");
        return 0;
    }
    int maxLength = 0;    // 记录最长不重复子串的长度
    int start = 0;        // 滑动窗口的起始位置
    int currentLength = 0; // 当前不重复子串的长度
    int startIndex = 0;   // 最长不重复子串的起始下标
    int endIndex = 0;     // 最长不重复子串的结束下标 
    // 用于记录字符是否出现过的哈希表
    // 假设输入的是ASCII字符，所以大小为128
    int charIndex[128];
    // 初始化哈希表，所有字符都初始化为-1（表示未出现过）
    for (int k = 0; k < 128; k++) {
        charIndex[k] = -1;
    }
    // 使用滑动窗口算法遍历字符串
    for (int i = 0; i < arrsize; i++) {
        char currentChar = arr[i];
        // 如果字符已经在当前窗口中出现过，移动窗口的起始位置
        if (charIndex[(unsigned char)currentChar] >= start) {
            start = charIndex[(unsigned char)currentChar] + 1;
        }
        // 更新当前字符的最后出现位置
        charIndex[(unsigned char)currentChar] = i;
        // 计算当前窗口的长度
        currentLength = i - start + 1;
        // 如果当前窗口长度大于最大长度，更新最大长度和子串位置
        if (currentLength > maxLength) {
            maxLength = currentLength;
            startIndex = start;
            endIndex = i;
        }
    }
    // 输出结果
    printf("最长的不重复字符串为%d个单位\n", maxLength);
    printf("最长不重复子串为：");
    for (int i = startIndex; i <= endIndex; i++) {
        printf("%c", arr[i]);
    }
    printf("\n");
    return maxLength;
}
