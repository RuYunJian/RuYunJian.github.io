#include <stdio.h>
int main()
{
int a[10];
int i = 0;
int k = 0; //交换中值
int n = 0;
int m = 0;
for (i = 0; i < 10; i++)
{
scanf_s("%d", &a[i]);
}
while (n < 9)
{
for (m = 0; m < 9-n; m++)
{
if (a[m] > a[m + 1])
{
k = a[m];
a[m] = a[m + 1];
a[m + 1] = k;
}
}
n++;
}
printf("排序后为： ");
for (int x = 0; x < 10; x++)
printf("%d ", a[x]);

}
